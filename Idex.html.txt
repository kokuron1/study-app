        body { font-family: sans-serif; background: #f0f2f5; padding: 15px; color: #333; }
        .card { background: white; border-radius: 20px; padding: 20px; box-shadow: 0 4px 15px rgba(0,0,0,0.1); }
        .bar { background: #eee; border-radius: 10px; height: 12px; margin: 15px 0; overflow: hidden; }
        .fill { background: #0088cc; height: 100%; width: 0%; transition: 0.4s; }
        .item { display: flex; align-items: center; justify-content: space-between; padding: 12px 0; border-bottom: 1px solid #f0f0f0; }
        .btn { background: #0088cc; color: white; border: none; width: 100%; padding: 15px; border-radius: 12px; font-weight: bold; margin-top: 20px; cursor: pointer; }

<body>
    <div class="card">
        <h2 id="title">Загрузка...</h2>
        <div class="bar"><div id="fill" class="fill"></div></div>
        <p>Прогресс: <span id="perc">0%</span></p>
        <div id="list"></div>
        <button class="btn" onclick="send()">💾 Сохранить прогресс</button>
    </div>

    <script>
        const tg = window.Telegram.WebApp;
        const params = new URLSearchParams(window.location.search);
        
        let data = {
            subject: params.get('sub') || "План не найден",
            topics: params.get('top') ? JSON.parse(params.get('top')) : []
        };

        function render() {
            const list = document.getElementById('list');
            list.innerHTML = '';
            let doneCount = 0;
            data.topics.forEach((t, i) => {
                if(t.done) doneCount++;
                list.innerHTML += `<div class="item"><span>${t.name}</span><input type="checkbox" ${t.done?'checked':''} onchange="tgl(${i})"></div>`;
            });
            let p = data.topics.length ? Math.round(doneCount/data.topics.length*100) : 0;
            document.getElementById('fill').style.width = p + '%';
            document.getElementById('perc').innerText = p + '%';
            document.getElementById('title').innerText = data.subject;
        }

        function tgl(i) { data.topics[i].done = !data.topics[i].done; render(); }
        function send() { tg.sendData(JSON.stringify(data)); tg.close(); }
        render();
    </script>
</body>
